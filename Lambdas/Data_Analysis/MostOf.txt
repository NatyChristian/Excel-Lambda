Nom : MostOf
Formule : 
LAMBDA(rng,[ScanByCols],[Ignore],[OnlyFirst],[LabelFilter],[ValFilter],                         LET(                                 SbC,IF(ISOMITTED(ScanByCols),0,ScanByCols),                                 Ig,IF(ISOMITTED(Ignore),0,Ignore),                                 lf,IF(ISOMITTED(LabelFilter),1,LabelFilter),                                  vf,IF(ISOMITTED(ValFilter),-1,ValFilter),                                 rw,TOCOL(rng,Ig,SbC),                                 Urw,UNIQUE(rw),                                 o,MEDIAN(IF(ISOMITTED(OnlyFirst),ROWS(Urw),OnlyFirst),ROWS(Urw),1),                                 cnt,MAP(Urw,LAMBDA(a,SUM(--(a=rw)))),                                 tab,HSTACK(Urw,cnt),                                 Stab,SORTBY(tab,cnt,vf,Urw,lf),                                  out,INDEX(Stab,SEQUENCE(o),SEQUENCE(,2)),out))
